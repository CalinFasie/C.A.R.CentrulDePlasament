CREATE TABLE `Utilizatori` (
	`id`	INTEGER NOT NULL DEFAULT 1 PRIMARY KEY AUTOINCREMENT UNIQUE,
	`nume`	VARCHAR,
	`parola`	VARCHAR
);

CREATE TABLE `luni` (
	`id_luni`	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE,
	`an_luni`	INTEGER,
	`luna_luni`	INTEGER
);

CREATE TABLE `Cotizatii` (
	`an`	INTEGER,
	`luna`	INTEGER,
	`nume`	TEXT NOT NULL,
	`marca`	INTEGER NOT NULL,
	`cnp`	INTEGER,
	`retinere`	INTEGER NOT NULL DEFAULT 0,
	`id`	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE,
	`dobanda`	INTEGER NOT NULL DEFAULT 0,
	`rata`	INTEGER NOT NULL DEFAULT 0,
	`fond`	INTEGER NOT NULL DEFAULT 0,
	`taxa`	INTEGER NOT NULL DEFAULT 0,
	`id_luna`	INTEGER,
	FOREIGN KEY(`id_luna`) REFERENCES luni ( id_luni )
);

CREATE TABLE `Manevra` (
	`an`	INTEGER NOT NULL,
	`luna`	INTEGER NOT NULL,
	`nume`	TEXT NOT NULL,
	`marca`	INTEGER NOT NULL,
	`cnp`	INTEGER,
	`retinere`	INTEGER NOT NULL DEFAULT 0,
	`id`	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE,
	`dobanda`	INTEGER NOT NULL DEFAULT 0,
	`rata`	INTEGER NOT NULL DEFAULT 0,
	`fond`	INTEGER NOT NULL DEFAULT 0,
	`taxa`	INTEGER NOT NULL DEFAULT 0,
	`id_luna`	INTEGER NOT NULL
);

CREATE TABLE `Print` (
	`an`	INTEGER NOT NULL,
	`luna`	INTEGER NOT NULL,
	`nume`	TEXT NOT NULL,
	`marca`	INTEGER NOT NULL,
	`cnp`	INTEGER,
	`retinere`	INTEGER NOT NULL DEFAULT 0,
	`id`	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE,
	`dobanda`	INTEGER NOT NULL DEFAULT 0,
	`rata`	INTEGER NOT NULL DEFAULT 0,
	`fond`	INTEGER NOT NULL DEFAULT 0,
	`taxa`	INTEGER NOT NULL DEFAULT 0,
	`id_luna`	INTEGER NOT NULL
);
